  function renderStore() {
    products.forEach((product, index) => {
      const item = document.createElement('div');
      item.className = 'item';
      item.innerHTML = `
        <img src="${product.image}" alt="${product.name}">
        <h2>${product.name}</h2>
        <p>${product.price} JOD</p>
        <p>${product.description}</p>
        <button class="cart-button" onclick="addToCart(${index})">Add to Cart</button>
      `;
      store.appendChild(item);
    });
  }

  function addToCart(index) {
    cart.push(products[index]);
    renderCart();
  }

  function renderCart() {
    cartItems.innerHTML = '';
    let total = 0;
    cart.forEach((item) => {
      const li = document.createElement('li');
      li.textContent = `${item.name} - ${item.price} JOD`;
      cartItems.appendChild(li);
      total += item.price;
    });
    totalDisplay.textContent = total;
  }

  function checkout() {
    let message = 'Hi! I would like to order the following items:%0A';
    let total = 0;
    cart.forEach(item => {
      message += `- ${item.name} (${item.price} JOD)%0A`;
      total += item.price;
    });
    message += `%0ATotal: ${total} JOD`;
    const url = `https://instagram.com/rjminis`; // Instagram doesn't support prefilled messages via link
    alert("Please screenshot your cart and send it via Instagram DM: @rjminis");
    window.open(url, '_blank');
